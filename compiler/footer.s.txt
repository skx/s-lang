        #
        # Exit explicitly, just in case the program was missing a
        # terminating RETURN statement.
        #
        xor rax, rax
        jmp exit_with_status



#####
#####  Standard Library
#####

        #
        # Return the length of the given string
        #

strlen:
        push rbp
        mov rbp, rsp

        mov rdi, [rbp+16]    # Get the string
        and rdi, -4          # Remove typing

        xor rax, rax
.strlen_next_char:
        cmp byte ptr [rdi + rax], 0
        je .strlen_done
        inc rax
        jmp .strlen_next_char
.strlen_done:
        shl rax, 2   # Add typing
        mov rsp, rbp
        pop rbp
        ret



        #
        # Compare strings for equality.
        #
        # Return zero if equal
        #
strcmp:
        push rbp
        mov rbp, rsp

        mov rdi, [rbp+16]    # Get the string
        and rdi, -4          # Remove typing
        mov rsi, [rbp+24]    # Get the string
        and rsi, -4          # Remove typing

        xor rax, rax
.strcmp_next_char:
        mov al, [rdi]
        cmp al, [rsi]
        jne .strcmp_diff
        test al, al
        je .strcmp_done
        inc rdi
        inc rsi
        jmp .strcmp_next_char
.strcmp_diff:
        sub rax, [rsi]
.strcmp_done:
        movsx rax, al   # Truncate, but keep the sign
        shl rax, 2   # Add typing
        mov rsp, rbp
        pop rbp
        ret




        #
        # Write a newline to STDOUT.
        #
newline:
        mov rax, 1            # sys_write
        mov rdi, 1            # stdout
        mov rsi, offset newline_buffer
        mov rdx, 1            # length = 1
        syscall
        ret

        #
        # Type will return a string which describes
        # the type of a variable
        #
type:
        # Mask off the lower two bits, to get the type info
        mov rcx, rax
        and rcx, 3

        cmp rcx, 0
        je type_is_int

        cmp rcx, 1
        je type_is_string

        cmp rcx, 2
        je type_is_float

        cmp rcx, 3
        je type_is_reserved
        ret

type_is_int:
        mov rax, offset type_is_int_str
        or rax, 1
        ret

type_is_string:
        mov rax, offset type_is_string_str
        or rax, 1
        ret

type_is_float:
        mov rax, offset type_is_float_str
        or rax, 1
        ret

type_is_reserved:
        mov rax, offset type_is_reserved_str
        or rax, 1
        ret


        #
        # Print will use the bottom bits of RAX to
        # determine type, and call the appropriate routine
        #
print:
        mov rcx, rax
        and rcx, 3

        cmp rcx, 0
        je print_integer

        cmp rcx, 1
        je print_string

        cmp rcx, 2
        je print_float

        cmp rcx, 3
        je print_reserved
        ret

print_float:
        lea rax, float_stub_msg
        or rax, 1
        jmp print_string

print_reserved:
        lea rax, reserved_stub_msg
        or rax, 1
        jmp print_string


        #
        # Print the NULL-terminated string pointed to by RAX.
        #
        # The length will be dynamically discovered
print_string:
        and rax, -4  # Undo typing
        mov rbx, rax

.loop:
        mov al, [rbx]         # load one byte
        test al, al           # is it zero?
        jz .done

        push rbx
        mov rax, 1            # sys_write
        mov rdi, 1            # stdout
        mov rsi, rbx          # pointer to char
        mov rdx, 1            # length = 1
        syscall
        pop rbx
        inc rbx               # next character
        jmp .loop
.done:
        ret


        #
        # Exit with the given status.
        #
        # Status code to use is stored in RAX
        #
exit_with_status:
        sar rax, 2  # undo typing
        mov rdi, rax
        mov rax, 60     # sys_exit
        syscall
        ret



#
# Convert the integer in RAX into
# ASCII and print it to the console.
#
# Uses the "print_rax_buffer" as temporary
# storage, and will trash it.
#
# Clobbers: rax, rbx, rcx, rdx, rdi, rsi
#
print_integer:
        sar rax, 2  # undo typing
        mov rbx, 10
        lea rdi, [print_rax_buffer+31]
        mov byte ptr [rdi], 0
        dec rdi

        # Track sign in rcx
        xor rcx, rcx

        # If negative:
        test rax, rax
        jns .convert_loop_start

        neg rax
        mov rcx, 1              # remember number was negative

.convert_loop_start:

        # Special case for 0
        test rax, rax
        jnz .convert_loop

        mov byte ptr [rdi], '0'
        dec rdi
        jmp .after_convert

.convert_loop:
        xor rdx, rdx
        div rbx
        add dl, '0'
        mov [rdi], dl
        dec rdi
        test rax, rax
        jnz .convert_loop

.after_convert:

        # Add minus sign if needed
        test rcx, rcx
        jz .done_sign

        mov byte ptr [rdi], '-'
        dec rdi

.done_sign:

        inc rdi              # rdi = pointer to string start

        # compute length in rdx
        lea rsi, [print_rax_buffer+31]
        mov rdx, rsi
        sub rdx, rdi

        mov rax, 1           # sys_write
        mov rsi, rdi         # buffer
        mov rdi, 1           # stdout
        syscall

        ret


        # Writeable data storage
        .section .data

        .align 4
newline_buffer:
        .ascii "\n"


        .align 4
float_stub_msg:
        .ascii "placeholder for FLOAT print\n"
        .byte 00

        .align 4
reserved_stub_msg:
        .ascii "placeholder for RESERVED print\n"
        .byte 00

        # Type strings - as returned by a call to type/type()
        .align 4
type_is_int_str:
        .ascii "INTEGER"
        .byte 00
        .align 4
type_is_float_str:
        .ascii "FLOAT"
        .byte 00
        .align 4
type_is_string_str:
        .ascii "STRING"
        .byte 00
        .align 4
type_is_reserved_str:
        .ascii "RESERVED"
        .byte 00

        .section .data

        # Buffer for printing numbers, and newlines
        .align 16
print_rax_buffer:
        .skip 32
